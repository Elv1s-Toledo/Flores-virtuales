  <script>

    /* Animación de flores sobre fondo galáctico usando Canvas */

    const canvas = document.getElementById('c');

    const ctx = canvas.getContext('2d', { alpha: true });

    let W = canvas.width = innerWidth;

    let H = canvas.height = innerHeight;

    window.addEventListener('resize', () => {

      W = canvas.width = innerWidth;

      H = canvas.height = innerHeight;

    });

    // utilidades

    const rand = (a,b)=> Math.random()*(b-a)+a;

    const TAU = Math.PI*2;

    // estrellas con suavizado

    const stars = Array.from({length:130}, ()=>({

      x: Math.random()*W,

      y: Math.random()*H,

      r: Math.random()*1.5+0.2,

      a: Math.random()*0.9+0.1,

      tw: Math.random()*0.02+0.002

    }));

    // Flores: cada flor tiene posición, tamaño, velocidad, color y rotación

    class Flower{

      constructor(x,y,scale=1){

        this.x = x;

        this.y = y;

        this.scale = scale * rand(0.7,1.2);

        this.vy = rand(-0.15,-0.6);

        this.vx = rand(-0.25,0.25);

        this.rotation = rand(0,TAU);

        this.vr = rand(-0.003,0.003);

        this.age = 0;

        this.life = rand(10,26); // segundos

        this.petals = Math.floor(rand(5,8));

        this.centerColor = '#e1a800';

        // amarillo de pétalos (ligeramente variado)

        const hue = Math.floor(rand(45,55));

        this.petalsColor = `hsl(${hue}deg ${90}% ${55}%)`;

        this.opacity = 0;

      }

      update(dt){

        this.age += dt;

        this.x += this.vx * dt * 60;

        this.y += this.vy * dt * 60;

        this.rotation += this.vr * dt * 60;

        // fade in/out

        if(this.age < 1.2) this.opacity = this.age/1.2;

        else if(this.age > this.life-1.4) this.opacity = Math.max(0,(this.life - this.age)/1.4);

        else this.opacity = 1;

      }

      draw(ctx){

        ctx.save();

        ctx.translate(this.x,this.y);

        ctx.rotate(this.rotation);

        ctx.globalAlpha = this.opacity * 0.95;

        ctx.scale(this.scale,this.scale);

        // Draw petals

        for(let i=0;i<this.petals;i++){

          const a = (i/this.petals)*TAU;

          ctx.save();

          ctx.rotate(a);

          ctx.beginPath();

          // petal shape using bezier

          ctx.moveTo(0,0);

          ctx.bezierCurveTo(20, -6, 48, -6, 72, 0);

          ctx.bezierCurveTo(48, 22, 20, 16, 0,0);

          ctx.closePath();

          // gradient for petals

          const g = ctx.createLinearGradient(10,0,72,0);

          g.addColorStop(0,'rgba(255,240,120,0.98)');

          g.addColorStop(0.5,this.petalsColor);

          g.addColorStop(1,'rgba(220,160,20,0.95)');

          ctx.fillStyle = g;

          ctx.fill();

          ctx.restore();

        }

        // center

        ctx.beginPath();

        ctx.arc(0,0,18,0,TAU);

        const gc = ctx.createRadialGradient(-6,-6,4,0,0,22);

        gc.addColorStop(0,'#fff6c8');

        gc.addColorStop(0.5,'#ffd54a');

        gc.addColorStop(1,this.centerColor);

        ctx.fillStyle = gc;

        ctx.fill();

        // subtle outline

        ctx.lineWidth = 1.2; ctx.strokeStyle = 'rgba(0,0,0,0.12)';

        ctx.stroke();

        ctx.restore();

      }

    }

    const flowers = [];

    // spawn cerca del centro o en la parte baja

    function spawnFlower(x,y,scale){

      flowers.push(new Flower(x ?? rand(W*0.2,W*0.8), y ?? rand(H*0.6,H*0.95), scale ?? rand(0.6,1.4)));

    }

    // evento: click para crear flor

    canvas.addEventListener('pointerdown', (e)=>{

      const rect = canvas.getBoundingClientRect();

      spawnFlower(e.clientX - rect.left, e.clientY - rect.top, rand(0.7,1.3));

    });

    // crear unas flores iniciales

    for(let i=0;i<8;i++) spawnFlower(rand(W*0.2,W*0.8), rand(H*0.6,H*0.95));

    // tiempo

    let last = performance.now();

    let running = true;

    document.getElementById('toggle').addEventListener('click', ()=>{

      running = !running; document.getElementById('toggle').textContent = running ? 'Pausar' : 'Reanudar';

      if(running) last = performance.now();

    });

    document.getElementById('spawn').addEventListener('click', ()=> spawnFlower());

    function drawBackground(ctx, t){

      // ligera nebulosa con gradientes movibles

      const g = ctx.createRadialGradient(W*0.8, H*0.2, 0, W*0.6, H*0.4, Math.max(W,H));

      g.addColorStop(0, 'rgba(120,70,180,0.06)');

      g.addColorStop(0.18,'rgba(40,20,90,0.06)');

      g.addColorStop(0.4,'rgba(20,40,60,0.04)');

      ctx.fillStyle = g; ctx.fillRect(0,0,W,H);

      // draw faint swirls (simple moving arcs)

      ctx.save();

      ctx.globalAlpha = 0.06;

      ctx.translate(W/2 + Math.sin(t*0.0003)*80, H/2 + Math.cos(t*0.0004)*60);

      ctx.beginPath();

      ctx.ellipse(0,0, W*0.6, H*0.35, Math.sin(t*0.0001)*0.6, 0, TAU);

      ctx.fillStyle = 'rgba(90,120,200,0.03)'; ctx.fill();

      ctx.restore();

      // estrellas drawn from array

      for(const s of stars){

        // twinkle

        const a = 0.5 + Math.sin(t*s.tw + s.x*0.001)*0.5;

        ctx.beginPath();

        ctx.globalAlpha = s.a * a * 0.8;

        ctx.arc(s.x, (s.y + Math.sin(t*0.0004 + s.x*0.0012)*6), s.r,0,TAU);

        ctx.fillStyle = '#ffffff'; ctx.fill();

      }

      ctx.globalAlpha = 1;

    }

    function loop(now){

      const dt = Math.min(0.06, (now - last)/1000);

      last = now;

      if(!running){ requestAnimationFrame(loop); return; }

      // clear with slight vignette to keep galaxy feel

      ctx.clearRect(0,0,W,H);

      // background slight overlay

      drawBackground(ctx, now);

      // update and draw flowers

      for(let i = flowers.length-1; i>=0; i--){

        const f = flowers[i];

        f.update(dt);

        f.draw(ctx);

        // remove if dead or out of bounds

        if(f.age > f.life || f.y < -120 || f.x < -200 || f.x > W+200) flowers.splice(i,1);

      }

      // floating small orbs like pollen

      // (simple particle effect)

      requestAnimationFrame(loop);

    }

    requestAnimationFrame(loop);

    // spawn periodic flowers

    setInterval(()=>{ if(running && flowers.length < 30) spawnFlower(rand(W*0.25,W*0.75), H+20, rand(0.7,1.5)); }, 1200);

    // accessibility: pause when page not visible

    document.addEventListener('visibilitychange', ()=>{ running = !document.hidden; document.getElementById('toggle').textContent = running ? 'Pausar' : 'Reanudar'; if(running) last = performance.now(); });

  </script>
